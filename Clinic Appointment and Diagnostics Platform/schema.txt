

patients[icon:user,color:blue]{
  patient_id serial pk
  name varchar(100) not null
  email varchar(100) unique not null
  phone varchar(15)
  date_of_birth date
  gender varchar(20)
  address text
  created_at timestamp
}

doctors[icon:user-check,color:green]{
  doctor_id serial pk
  name varchar(100) not null
  email varchar(100) unique not null
  phone varchar(15)
  specialization varchar(100)
  experience_years int
}

appointments[icon:calendar,color:orange]{
  appointment_id serial pk
  patient_id int fk
  doctor_id int fk
  appointment_date date
  time_slot varchar(50)
  status enum['BOOKED','CANCELLED','COMPLETED','NO_SHOW']
  created_at timestamp
}

consultations[icon:stethoscope,color:red]{
  consultation_id serial pk
  appointment_id int fk
  patient_id int fk
  doctor_id int fk
  consultation_date timestamp
  diagnosis text
  notes text
}

tests[icon:activity,color:purple]{
  test_id serial pk
  test_name varchar(100)
  description text
  cost int
}

consultation_tests[icon:link,color:pink]{
  id serial pk
  consultation_id int fk
  test_id int fk
  instructions text
}

reports[icon:file-text,color:yellow]{
  report_id serial pk
  consultation_id int fk
  test_id int fk
  result text
  report_date date
  file_path text
}

payments[icon:credit-card,color:black]{
  payment_id serial pk
  patient_id int fk
  appointment_id int fk
  amount int
  payment_method enum['UPI','CARD','CASH']
  status enum['PENDING','SUCCESS','FAILED']
  payment_date timestamp
}


patients.patient_id < appointments.patient_id
doctors.doctor_id < appointments.doctor_id

appointments.appointment_id - consultations.appointment_id
patients.patient_id < consultations.patient_id
doctors.doctor_id < consultations.doctor_id

consultations.consultation_id < consultation_tests.consultation_id
tests.test_id < consultation_tests.test_id

consultations.consultation_id < reports.consultation_id
tests.test_id < reports.test_id

patients.patient_id < payments.patient_id
appointments.appointment_id < payments.appointment_id
