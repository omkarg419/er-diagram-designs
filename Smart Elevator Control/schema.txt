buildings[icon:building,color:blue]{
  building_id serial pk
  building_name varchar(100)
  location varchar(150)
  total_floors int
  created_at timestamp
}

floors[icon:layers,color:cyan]{
  floor_id serial pk
  building_id int fk
  floor_number int
}

elevator_shafts[icon:box,color:gray]{
  shaft_id serial pk
  building_id int fk
  shaft_number int
}

elevators[icon:arrow-up-down,color:green]{
  elevator_id serial pk
  building_id int fk
  shaft_id int fk
  elevator_number varchar(20)
  capacity int
}

elevator_floor_map[icon:link,color:purple]{
  id serial pk
  elevator_id int fk
  floor_id int fk
}

floor_requests[icon:bell,color:orange]{
  request_id serial pk
  floor_id int fk
  request_time timestamp
  direction enum['UP','DOWN']
  status enum['PENDING','ASSIGNED','COMPLETED']
}

ride_assignments[icon:shuffle,color:red]{
  assignment_id serial pk
  request_id int fk
  elevator_id int fk
  assigned_time timestamp
}

ride_logs[icon:activity,color:pink]{
  ride_id serial pk
  elevator_id int fk
  source_floor_id int fk
  destination_floor_id int fk
  start_time timestamp
  end_time timestamp
}

elevator_status[icon:cpu,color:yellow]{
  status_id serial pk
  elevator_id int fk
  status enum['IDLE','MOVING','MAINTENANCE','OUT_OF_SERVICE']
  updated_at timestamp
}

maintenance_records[icon:tool,color:black]{
  maintenance_id serial pk
  elevator_id int fk
  issue_description text
  start_time timestamp
  end_time timestamp
  status enum['OPEN','IN_PROGRESS','COMPLETED']
}

buildings.building_id < floors.building_id
buildings.building_id < elevators.building_id
buildings.building_id < elevator_shafts.building_id

elevator_shafts.shaft_id < elevators.shaft_id

elevators.elevator_id < elevator_floor_map.elevator_id
floors.floor_id < elevator_floor_map.floor_id

floors.floor_id < floor_requests.floor_id

floor_requests.request_id - ride_assignments.request_id
elevators.elevator_id < ride_assignments.elevator_id

elevators.elevator_id < ride_logs.elevator_id
floors.floor_id < ride_logs.source_floor_id
floors.floor_id < ride_logs.destination_floor_id

elevators.elevator_id < elevator_status.elevator_id
elevators.elevator_id < maintenance_records.elevator_id
